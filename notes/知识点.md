1.urlsafe_b64encode
在base64的基础上替换了不安全的一些字符
	urlsafe_b64encode = function(input) {
		return base64.encoder(input).replace(/\+/g, '-').replace(/\//g, '_');
	};

用'-'替代'+'，用'_'替代'/'
